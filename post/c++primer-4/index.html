    //while形式

    //for形式
    for(init-statement;codition;expression)
        statement;

    //for语句头的多重定义，注意for语句内定义的变量只在for块内有效，出了for块就会被销毁
    for(decltype(v.size()) i = 0, sz = v.size();i != sz;++i)
        v.push_back(v[i]);
    //for语句可以省略部分语句，但是要带分号

    //c++11新增加范围for语句
    for (declaration : expression)
        statement
    // do while语句,与while很相似，但是区别是先执行一次循环，再判断
    do
        statement
    while(condition);
