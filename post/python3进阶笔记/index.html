#python中的type、object、class的关系
#在python中，我们知道我们可以用type函数来打印某个对象的类型，如：
>>> a=1
>>> type(a)
<class 'int'>
>>> s='hello'
>>> type(s)
<class 'str'>
#那上面说类也是对象，那我们type类又会怎么样呢？
>>> type(int)
<class 'type'>
>>> type(str)
<class 'type'>
#?什么东西呀，int的类型是type类？int类是type类的实例？
#我们试试自定义的类
>>> type(Student)

#什么鬼？看起来类类型都是type类的实例,那type类又是谁的实例呢？
>>> type(type)

#？？？糊涂了吧，type类居然是它自己类的实例？

#-------------------------------------------------

>>> int.__bases__
(<class 'object'>,)
>>> str.__bases__
(<class 'object'>,)
>>> Student.__bases__
(<class 'object'>,)
#原来python里一切类的基类都是继承object的。嗯。那type类呢？
>>> type.__bases__

>>> type(object)
<class 'type'>#? 等等，有点晕，object的类型是type，object类是type的实例，而type的基类又是object的？搞什么飞机啊

#废话不多少，先来创建一个类
class School:
    def __init__(self,students_list):
        self.students = students_list

    #魔法函数一
    def __getitem__(self, item):
        return self.students[item]
    

    #魔法函数二
    def __len__(self):
        return len(self.students)
school = School(['李明','小明','小爱'])
#for循环是要作用于可迭代对象的，我们增加了getitem这个魔法函数竟然能使该对象变成可迭代对象，我的天呐，这么神奇吗？
#心想，难怪叫做魔法函数，果然有魔法。
for item in school:
    print(item)
#output:
#李明
#小明
#小爱

#看看其它操作，啥也不说了，牛🍺就完事了。
print(school[1])
print(school[1:])
#output：
#小明
#['小明', '小爱']

#看看魔法函数二：len,嗯可以用len来调用它的__len__方法，神奇
print(len(school))
#output:3
#添加魔法函数三之前
print(school)
#output:<__main__.School object at 0x102b62358>
#添加魔法函数三定制我们自己的对象输出格式
#output:李明,小明,小爱
#还有个__repr__魔法函数是显示元素字符的，不是print打印出来的字符。不多介绍了。

#python魔法函数还有很多，需要自己去尝试，后面的笔记可能会提到。
#len函数用于内置类型的话，速度是很快的，因为python解释器优化了这些这些东西，C语言写的解释器其中肯定包含了列表的大小。
